69 template<class Key, class Value> void EvictionSegmentedLRU<Key, Value>::on_cache_hit(const CacheItem& item)
70 {
71  assert(m_probation_nodes.size() == m_probation_list.size());
72  assert(m_protected_nodes.size() == m_protected_list.size());
73 
74  auto protected_node_it = m_protected_nodes.find(item.m_key);
75  if (protected_node_it != m_protected_nodes.end()) {
76  if (protected_node_it->second != m_protected_list.begin()) {
77  // If the node is in the protected segment, move it to the front of the protected segment.
78  m_protected_list.splice(m_protected_list.begin(), m_protected_list, protected_node_it->second);
79  }
80  } else {
81  // If the node is in probation, move it to the protected segment.
82  const bool promotion_ok = move_to_protected(item.m_key);
83  assert(promotion_ok);
84  }
85 
86  while (m_protected_list.size() > m_protected_segment_size) {
87  const bool demotion_ok = pop_to_probation();
88  assert(demotion_ok);
89  assert(m_protected_list.size() == m_protected_segment_size);
90  }
91 
92  assert(m_probation_nodes.size() == m_probation_list.size());
93  assert(m_protected_nodes.size() == m_protected_list.size());
94 }

96 template<class Key, class Value> void EvictionSegmentedLRU<Key, Value>::on_evict(const Key& key)
97 {
98  assert((m_probation_list.empty() && !m_protected_list.empty() && m_protected_list.back().get() == key) ||
99  (!m_probation_list.empty() && m_probation_list.back().get() == key));
100 
101  if (!m_probation_list.empty() && m_probation_list.back().get() == key) {
102  // There are items in the probation list, these should be evicted first.
103  assert(m_probation_nodes.find(key) != m_probation_nodes.end());
104  m_probation_nodes.erase(key);
105  m_probation_list.pop_back();
106  } else {
107  // The probation list is empty, evict from the protected list.
108  assert(m_protected_nodes.find(key) != m_protected_nodes.end());
109  m_protected_nodes.erase(key);
110  m_protected_list.pop_back();
111  }
