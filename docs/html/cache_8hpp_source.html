51 template<class K, class V, template<class, class> class I, template<class, class> class E, class SV, class SK>
52 bool Cache<K, V, I, E, SV, SK>::insert(const K& key, const V& value)
53 {
54  std::lock_guard<std::mutex> guard{m_mutex};
55 
56  const auto key_size = static_cast<size_t>(m_measure_key(key));
57  const auto value_size = static_cast<size_t>(m_measure_value(value));
58  const size_t item_size = key_size + value_size;
59  const size_t item_size_overhead = item_size + key_size;
60 
61  const bool should_insert = compare_evict(key, item_size_overhead);
62 
63  if (should_insert) {
64  auto key_and_item = m_data.find(key);
65  if (key_and_item != m_data.end()) {
66  // Update.
67 
68  if (value_size > key_and_item->second.m_value_size) {
69  // This item got bigger with the update.
70  m_current_size += value_size - key_and_item->second.m_value_size;
71  } else {
72  // This item got smaller with the update.
73  m_current_size -= key_and_item->second.m_value_size - value_size;
74  }
75 
76  key_and_item->second.m_value = value;
77  key_and_item->second.m_value_size = value_size;
78  key_and_item->second.m_total_size = item_size;
79 
80  on_update(key_and_item->second);
81  } else {
82  // Insert.
83  const auto it_and_ok =
84  m_data.emplace(std::piecewise_construct, std::forward_as_tuple(key), std::forward_as_tuple(key, key_size, value, value_size));
85  assert(it_and_ok.second);
86  const CacheItem& item = it_and_ok.first->second;
87  on_insert(item);
88 
89  m_current_size += item_size_overhead;
90  }
91  }
92 
93  return should_insert;
94 }

154 template<class K, class V, template<class, class> class I, template<class, class> class E, class SV, class SK>
155 bool Cache<K, V, I, E, SV, SK>::compare_evict(const K& candidate_key, size_t candidate_size)
156 {
157  if (m_current_size + candidate_size <= m_maximum_size) {
158  // We have enough room. Insert if the policy agrees.
159  return m_insertion_policy->should_add(candidate_key);
160  } else {
161  if (candidate_size > m_maximum_size) {
162  // The item to insert/update is bigger than the maximum size of the cache, no need to attempt an insertion.
163  return false;
164  }
165  // We need to get a list of keys to evict to make room for the candidate. Since we restrict cache inserts by
166  // cache size instead of by nb. of items, we might evict multiple items to insert one. This is bad because it
167  // leads to inaccuracies:
168  //
169  // Say for instance we want to insert a 4 kb item. In order to insert it,
170  // we might need to evict 4 1kb items.
171  // Since this is a feedback cache, insertion policies operate by comparing two keys (the candidate and an
172  // eviction victim) together and inserting/keeping the best one. This means that in order to insert our 4kb
173  // item, we'd need to make 4 calls to the insertion policy, and insert the item only if all 4 calls come out
174  // positive. *However*, this only establishes that the candidate is better than each individual replaced item,
175  // not that it is better than the sum of them... This could be solved in two ways.
176  //
177  // The easiest way to fix the issue would be to change the cache constraint from "allocated memory" to "number
178  // of cached items". This would lead to some inaccuracies in how memory is tracked, it but would
179  // increase the performance of the cache (an insert would always evict [0,1] item).
180  //
181  // The harder way would be to change the interface of the insertion policy. Instead of should_replace(Key&,
182  // Key&), we could do should_replace(Key&, vector<Key&>). This would allow the insertion policy to weigh the
183  // candidate vs. *all* the evicted keys. (e.g. an LFU policy could make sure that the load counts of the
184  // candidate needs to be higher than the _sum_ of load counts of all victims) I consider this method the harder
185  // one because it would make the implementation of all insertion policies considerably harder. (it's not always
186  // trivial to determine whether a key is a better candidate than a given set of keys)
187  //
188  // Those two ways could even be used together: we could change the insertion policy interface *and* provide a
189  // new count-limited cache for improved performance, while keeping the memory-limited cache for when required.
190 
191  std::vector<typename DataMap::iterator> keys_to_evict;
192  size_t freed_amount = 0;
193 
194  auto should_evict_another = [&](auto victim_it) { return victim_it != m_eviction_policy->victim_end() && freed_amount <= candidate_size; };
195 
196  for (auto victim_it = m_eviction_policy->victim_begin(); should_evict_another(victim_it); ++victim_it) {
197  const K& key_to_evict = *victim_it;
198  auto key_and_item = m_data.find(key_to_evict);
199 
200  if (key_and_item != m_data.end()) {
201  if (!m_insertion_policy->should_replace(key_to_evict, candidate_key)) {
202  return false;
203  }
204  const auto total_size_and_overhead = key_and_item->second.m_key_size + key_and_item->second.m_total_size;
205  freed_amount += total_size_and_overhead;
206  keys_to_evict.push_back(key_and_item);
207  } else {
208  // If this trips, the eviction policy tried to evict an item not in cache: the eviction policy and the
209  // cache are out of sync.
210  assert(false);
211  }
212  }
213 
214  // If this trips, something is very wrong with how size is tracked.
215  // This means that evicting all items didn't free enough space in the cache for the candidate,
216  // yet the check at the beginning of this method ensures that the new item is smaller than
217  // the max size of the cache...
218  assert((m_current_size - freed_amount) + candidate_size <= m_maximum_size);
219 
220  // Perform the eviction(s).
221  for (auto key_and_item : keys_to_evict) {
222  const auto total_size_and_overhead = key_and_item->second.m_key_size + key_and_item->second.m_total_size;
223  m_current_size -= total_size_and_overhead;
224 
225  on_evict(key_and_item->first);
226  m_data.erase(key_and_item->first);
227  }
228 
229  return true;
230  }
231 }
