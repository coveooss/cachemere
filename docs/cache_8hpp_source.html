56 void Cache<K, V, I, E, SV, SK, TS>::collect_into(C& container) const
57 {
58  using namespace boost;
59  using namespace detail;
60 
61  // Use emplace_back if container is a sequence container, or emplace if container is an associative container.
62  constexpr auto emplace_fn = hana::if_(
63  traits::stl::has_emplace_back<C, K, V>,
64  [](auto& seq_container, const auto& item) { seq_container.emplace_back(item.m_key, item.m_value); },

68 
69  // Reserve space if the container has a reserve() method and a size method().
70  hana::if_(

72  [&](auto& c) { c.reserve(c.size() + m_data.size()); },
73  [](auto&) {})(container);

76  for (const auto& [_, cached_item] : m_data) {

81 template<class K, class V, template<class, class> class I, template<class, class> class E, class SV, class SK, bool TS>
82 bool Cache<K, V, I, E, SV, SK, TS>::insert(const K& key, const V& value)
83 {
84  std::unique_lock<std::mutex> guard(lock());
85 
86  const auto key_size = static_cast<size_t>(m_measure_key(key));
87  const auto value_size = static_cast<size_t>(m_measure_value(value));
88  const size_t item_size = key_size + value_size;
89  const size_t item_size_overhead = item_size + key_size;
90 
91  const bool should_insert = compare_evict(key, item_size_overhead);
92 
93  if (should_insert) {
94  auto key_and_item = m_data.find(key);
95  if (key_and_item != m_data.end()) {
96  // Update.
97 
98  if (value_size > key_and_item->second.m_value_size) {
99  // This item got bigger with the update.
100  m_current_size += value_size - key_and_item->second.m_value_size;
101  } else {
102  // This item got smaller with the update.
103  m_current_size -= key_and_item->second.m_value_size - value_size;
104  }
105 
106  key_and_item->second.m_value = value;
107  key_and_item->second.m_value_size = value_size;
108  key_and_item->second.m_total_size = item_size;
109 
110  on_update(key_and_item->second);
111  } else {
112  // Insert.
113  const auto it_and_ok =
114  m_data.emplace(std::piecewise_construct, std::forward_as_tuple(key), std::forward_as_tuple(key, key_size, value, value_size));
115  assert(it_and_ok.second);
116  const CacheItem& item = it_and_ok.first->second;
117  on_insert(item);
118 
119  m_current_size += item_size_overhead;
120  }
121  }
122 
123  return should_insert;
124 }

250 template<class K, class V, template<class, class> class I, template<class, class> class E, class SV, class SK, bool TS>
251 bool Cache<K, V, I, E, SV, SK, TS>::compare_evict(const K& candidate_key, size_t candidate_size)
252 {
253  if (m_current_size + candidate_size <= m_maximum_size) {
254  // We have enough room. Insert if the policy agrees.
255  return m_insertion_policy->should_add(candidate_key);
256  } else {
257  if (candidate_size > m_maximum_size) {
258  // The item to insert/update is bigger than the maximum size of the cache, no need to attempt an insertion.
259  return false;
260  }
261  // We need to get a list of keys to evict to make room for the candidate. Since we restrict cache inserts by
262  // cache size instead of by nb. of items, we might evict multiple items to insert one. This is bad because it
263  // leads to inaccuracies:
264  //
265  // Say for instance we want to insert a 4 kb item. In order to insert it,
266  // we might need to evict 4 1kb items.
267  // Since this is a feedback cache, insertion policies operate by comparing two keys (the candidate and an
268  // eviction victim) together and inserting/keeping the best one. This means that in order to insert our 4kb
269  // item, we'd need to make 4 calls to the insertion policy, and insert the item only if all 4 calls come out
270  // positive. *However*, this only establishes that the candidate is better than each individual replaced item,
271  // not that it is better than the sum of them... This could be solved in two ways.
272  //
273  // The easiest way to fix the issue would be to change the cache constraint from "allocated memory" to "number
274  // of cached items". This would lead to some inaccuracies in how memory is tracked, it but would
275  // increase the performance of the cache (an insert would always evict [0,1] item).
276  //
277  // The harder way would be to change the interface of the insertion policy. Instead of should_replace(Key&,
278  // Key&), we could do should_replace(Key&, vector<Key&>). This would allow the insertion policy to weigh the
279  // candidate vs. *all* the evicted keys. (e.g. an LFU policy could make sure that the load counts of the
280  // candidate needs to be higher than the _sum_ of load counts of all victims) I consider this method the harder
281  // one because it would make the implementation of all insertion policies considerably harder. (it's not always
282  // trivial to determine whether a key is a better candidate than a given set of keys)
283  //
284  // Those two ways could even be used together: we could change the insertion policy interface *and* provide a
285  // new count-limited cache for improved performance, while keeping the memory-limited cache for when required.
286 
287  std::vector<DataMapIt> keys_to_evict;
288  size_t freed_amount = 0;
289 
290  auto should_evict_another = [&](auto victim_it) { return victim_it != m_eviction_policy->victim_end() && freed_amount <= candidate_size; };
291 
292  for (auto victim_it = m_eviction_policy->victim_begin(); should_evict_another(victim_it); ++victim_it) {
293  const K& key_to_evict = *victim_it;
294  auto key_and_item = m_data.find(key_to_evict);
295 
296  if (key_and_item != m_data.end()) {
297  if (!m_insertion_policy->should_replace(key_to_evict, candidate_key)) {
298  return false;
299  }
300  const auto total_size_and_overhead = key_and_item->second.m_key_size + key_and_item->second.m_total_size;
301  freed_amount += total_size_and_overhead;
302  keys_to_evict.push_back(key_and_item);
303  } else {
304  // If this trips, the eviction policy tried to evict an item not in cache: the eviction policy and the
305  // cache are out of sync.
306  assert(false);
307  }
308  }
309 
310  // If this trips, something is very wrong with how size is tracked.
311  // This means that evicting all items didn't free enough space in the cache for the candidate,
312  // yet the check at the beginning of this method ensures that the new item is smaller than
313  // the max size of the cache...
314  assert((m_current_size - freed_amount) + candidate_size <= m_maximum_size);
315 
316  // Perform the eviction(s).
317  for (auto key_and_item : keys_to_evict) {
318  remove(key_and_item);
319  }
320 
321  return true;
322  }
323 }
