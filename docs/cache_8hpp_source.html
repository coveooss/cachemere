64 template<class Container>
65 void Cache<K, V, I, E, C, SV, SK, TS>::collect_into(Container& container) const
66 {
67  using namespace boost;
68  using namespace detail;
69 
70  // Use emplace_back if container is a sequence container, or emplace if container is an associative container.
71  constexpr auto emplace_fn = hana::if_(
72  traits::stl::has_emplace_back<Container, K, V>,
73  [](auto& seq_container, const auto& key, const auto& item) { seq_container.emplace_back(key, item.m_value); },
74  [](auto& assoc_container, const auto& key, const auto& item) { assoc_container.emplace(key, item.m_value); });
75 
76  std::unique_lock<std::recursive_mutex> guard(lock());
77 
78  // Reserve space if the container has a reserve() method and a size method().
79  hana::if_(
80  hana::and_(traits::stl::has_reserve<Container>, traits::stl::has_size<Container>),
81  [&](auto& c) { c.reserve(c.size() + m_data.size()); },
82  [](auto&) {})(container);
83 
84  // Copy the cache contents to the container.
85  for (const auto& [key, cached_item] : m_data) {
86  emplace_fn(container, key, cached_item);
87  }
88 }

90 template<class K, class V, template<class, class> class I, template<class, class> class E, template<class, class> class C, class SV, class SK, bool TS>
91 bool Cache<K, V, I, E, C, SV, SK, TS>::insert(K key, V value)
92 {
93  std::unique_lock<std::recursive_mutex> guard(lock());
94 
95  const auto key_size = static_cast<size_t>(m_measure_key(key));
96  const auto value_size = static_cast<size_t>(m_measure_value(value));
97 
98  CacheItem new_item{key_size, std::move(value), value_size};
99 
100  auto it = m_data.find(key);
101  if (it != m_data.end()) {
102  if (check_replace(key, it->second, new_item)) {
103  // We call insert_or_update because we might have evicted the original key to make room for this one.
104  insert_or_update(std::move(key), std::move(new_item));
105  return true;
106  }
107  } else {
108  if (check_insert(key, new_item)) {
109  const auto it_and_ok = m_data.insert_or_assign(std::move(key), std::move(new_item));
110  assert(it_and_ok.second);
111 
112  on_insert(it_and_ok.first->first, it_and_ok.first->second);
113  return true;
114  }
115  }
116 
117  return false;
118 }

194 template<typename... Args>
195 void Cache<K, V, I, E, C, SV, SK, TS>::update_constraint(Args... args)
196 {
197  std::unique_lock<std::recursive_mutex> guard(lock());
198  m_constraint_policy->update(std::forward<Args>(args)...);
199 
200  auto should_evict_another = [&](auto victim_it) { return victim_it != m_eviction_policy->victim_end() && !m_constraint_policy->is_satisfied(); };
201 
202  for (auto victim_it = m_eviction_policy->victim_begin(); should_evict_another(victim_it); victim_it = m_eviction_policy->victim_begin()) {
203  const K& key_to_evict = *victim_it;
204  auto key_and_item = m_data.find(key_to_evict);
205 
206  if (key_and_item != m_data.end()) {
207  remove(key_and_item);
208  } else {
209  // If this trips, the eviction policy tried to evict an item not in cache: the eviction policy and the cache are out of sync.
210  assert(false);
211  }
212  }
213 
214  assert(m_constraint_policy->is_satisfied());

311 template<class K, class V, template<class, class> class I, template<class, class> class E, template<class, class> class C, class SV, class SK, bool TS>
312 bool Cache<K, V, I, E, C, SV, SK, TS>::check_insert(const K& key, const CacheItem& item)
313 {
314  if (m_constraint_policy->can_add(key, item)) {
315  return m_insertion_policy->should_add(key);
316  }
317 
318  // We need to perform some evictions to try and make some room.
319  // Since the insertion process can fail at anytime before we know how many keys to evict (e.g. if should_replace was to return false) however,
320  // we can't directly evict items as we go. To fix this, we copy the constraint policy to see how many keys we'd have to evict. If we manage to
321  // satisfy the constraint copy, we evict the keys we picked and proceed with the insertion.
322  auto constraint_copy = std::make_unique<MyConstraintPolicy>(*m_constraint_policy);
323  std::vector<DataMapIt> keys_to_evict;
324 
325  auto should_evict_another = [&](auto victim_it) { return victim_it != m_eviction_policy->victim_end() && !constraint_copy->can_add(key, item); };
326 
327  // As long as the constraint isn't satisfied, we keep evicting keys.
328  for (auto victim_it = m_eviction_policy->victim_begin(); should_evict_another(victim_it); ++victim_it) {
329  const K& key_to_evict = *victim_it;
330  auto key_and_item = m_data.find(key_to_evict);
331 
332  if (key_and_item != m_data.end()) {
333  if (!m_insertion_policy->should_replace(key_to_evict, key)) {
334  // This key to evict is considered "better" to have in cache than the item to be inserted.
335  // In this case, we abort the insertion.
336  return false;
337  }
338 
339  constraint_copy->on_evict(key_and_item->first, key_and_item->second);
340  keys_to_evict.push_back(key_and_item);
341  } else {
342  // If this trips, the eviction policy tried to evict an item not in cache: the eviction policy and the
343  // cache are out of sync.
344  assert(false);
345  }
346  }
347 
348  if (constraint_copy->can_add(key, item)) {
349  // The constraint is happy with that, so we actually evict the collected keys.
350  for (auto key_and_item : keys_to_evict) {
351  remove(key_and_item);
352  }
353  return true;
354  }
355 
356  return false;
357 }

359 template<class K, class V, template<class, class> class I, template<class, class> class E, template<class, class> class C, class SV, class SK, bool TS>
360 bool Cache<K, V, I, E, C, SV, SK, TS>::check_replace(const K& key, const CacheItem& old_item, const CacheItem& new_item)
361 {
362  if (m_constraint_policy->can_replace(key, old_item, new_item)) {
363  return true;
364  }
365 
366  // The logic here is very similar to check_insert but with an important modification.
367  // Since in this code path we're updating an existing key instead of adding a new one, we need to handle the case
368  // where the eviction policy recommends the eviction of the key we're trying to insert. If this happens and the constraint
369  // is still not satisfied afterwards, we need to treat all subsequent constraint checks as if we were inserting instead of updating our key
370  // (because the original was evicted).
371  auto constraint_copy = std::make_unique<MyConstraintPolicy>(*m_constraint_policy);
372  bool evicted_original_key = false;
373 
374  auto can_replace = [&]() {
375  if (evicted_original_key) {
376  return constraint_copy->can_add(key, new_item);
377  } else {
378  return constraint_copy->can_replace(key, old_item, new_item);
379  }
380  };
381 
382  auto should_evict_another = [&](auto victim_it) { return victim_it != m_eviction_policy->victim_end() && !can_replace(); };
383 
384  // Evict until the constraint copy is happy.
385  std::vector<DataMapIt> keys_to_evict;
386 
387  for (auto victim_it = m_eviction_policy->victim_begin(); should_evict_another(victim_it); ++victim_it) {
388  const K& key_to_evict = *victim_it;
389  auto key_and_item = m_data.find(key_to_evict);
390 
391  if (key_and_item != m_data.end()) {
392  if (!m_insertion_policy->should_replace(key_to_evict, key)) {
393  return false;
394  }
395 
396  if (!evicted_original_key) {
397  evicted_original_key = key_and_item->first == key;
398  }
399 
400  constraint_copy->on_evict(key_and_item->first, key_and_item->second);
401  keys_to_evict.push_back(key_and_item);
402  } else {
403  // If this trips, the eviction policy tried to evict an item not in cache: the eviction policy and the
404  // cache are out of sync.
405  assert(false);
406  }
407  }
408 
409  if (can_replace()) {
410  for (auto key_and_item : keys_to_evict) {
411  remove(key_and_item);
412  }
413  return true;
414  }
415 
416  return false;
417 }

483 void Cache<K, V, I, E, C, SV, SK, TS>::on_cache_hit(const K& key, const CacheItem& item) const
484 {
485  // Update the cache hit rate accumulators.
486  m_hit_rate_acc(1);
487  m_byte_hit_rate_acc(static_cast<uint32_t>(item.m_value_size));
488 
489  // Call event handler iif the method is defined in the policy.
490  boost::hana::if_(
